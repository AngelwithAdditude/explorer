 19 define(['dojo/_base/declare', 'modules/widgets/controlgroups/ControlGroup',
 20         'dojo/query', 'dojo/dom-construct', 'dojo/NodeList-manipulate', 'dojo/NodeList-dom'],
 21         function(declare, ControlGroup, query, domConstruct, domClass) {
 22             return declare('StringControlGroupWidget', [ ControlGroup ], {
 23                 
 24                 postCreate : function() {
 25                   this.input = domConstruct.create("input", {"type" : "text", "value" : this.defaultValue});
 26                   query('.controls', this.domNode).innerHTML(this.input);
 27                 },
 28                 
 29                 getValue : function() {
 30                   return this.input.value;
 31                 },
 32                 
 33                 setValue : function(value) {
 34                   this.input.value = value;
 35                 }
 36             });
 37         });
