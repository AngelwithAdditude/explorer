 19 define(['dojo/_base/declare', 'dijit/_WidgetBase', 'dijit/_TemplatedMixin',
 20         'dojo/text!./../../templates/GadgetMenuButton.html',
 21         'modules/widgets/gadgetarea/GadgetDropDownMenu', 'dojo/dom-construct'],
 22         function(declare, WidgetBase, TemplatedMixin, template, GadgetDropDownMenu, domConstruct) {
 23             return declare('GadgetMenuButtonWidget', [ WidgetBase, TemplatedMixin ], {
 24                 templateString : template,
 25 
 26                 startup : function() {
 27                   this.gadgetDropDown = new GadgetDropDownMenu({"gadgetArea" : this.gadgetArea, "prefDialog" : this.prefDialog});
 28                   domConstruct.place(this.gadgetDropDown.domNode, this.domNode);
 29                   this.gadgetDropDown.startup();
 30                 },
 31                 
 32                 constructMenu : function(metadata) {
 33                   this.gadgetDropDown.setViews(metadata.views);
 34                 },
 35                 
 36                 getGadgetDropDownMenu : function() {
 37                   return this.gadgetDropDown;
 38                 }
 39             });
 40         });
