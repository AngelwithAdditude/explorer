 19 define(['dojo/_base/declare', 'dijit/_WidgetBase', 'dijit/_TemplatedMixin',
 20         'dojo/_base/array', 'dojo/text!./../../templates/GadgetArea.html', 'modules/widgets/gadgetarea/GadgetToolbar',
 21         'dojo/dom-construct','modules/widgets/Loading', 'modules/opensocial-data', 'modules/widgets/gadgetarea/GadgetModalDialog',
 22         'dojo/_base/window', 'dojo/dom', 'dojo/json'],
 23         function(declare, WidgetBase, TemplatedMixin,
 24                 arrayUtil, template, GadgetToolbar, domConstruct, Loading, osData, GadgetModalDialog, win, dom,
 25                 JSON) {
 26             var GadgetArea = declare('GadgetAreaWidget', [ WidgetBase, TemplatedMixin ], {
 27                 templateString : template,
 28                 
 29                 constructor : function() {
 30                   this.siteCounter = 0;
 31                   
 32                   var config = {},
 33                       self = this,
 34                       lifecycle = {};
 35                   config[osapi.container.ContainerConfig.RENDER_DEBUG] = '1';
 36                   config[osapi.container.ContainerConfig.SET_PREFERENCES] = this.setPrefs();
 37                   this.container = new osapi.container.Container(config);
 38                   lifecycle[osapi.container.CallbackType.ON_RENDER] = function(gadgetUrl, sideId) {
 39                     self.loadingWidget.hide();
 40                   };
 41                   this.container.addGadgetLifecycleCallback('org.opensocial.explorer', lifecycle);
 42                   
 43                   // Hook-up actions
 44                   this.container.actions.registerShowActionsHandler(function(actionObjArray) { 
 45                     self.showActions(actionObjArray, self.gadgetToolbar, self.container, function(actionId){
 46                       self.container.actions.runAction(actionId);
 47                     });
 48                   });
 49                   this.container.actions.registerHideActionsHandler(function(actionObjArray) { 
 50                     self.hideActions(actionObjArray, self.gadgetToolbar);
 51                   });
 52                   this.container.actions.registerNavigateGadgetHandler(function(gadgetUrl, opt_params) {
 53                     self.navigateForActions(self, gadgetUrl, opt_params);
 54                   });
 55                   
 56                   //Hook-up open-views
 57                   this.container.views.createElementForUrl = this.handleNavigateUrl();
 58                   this.container.views.createElementForGadget = this.handleNavigateGadget();
 59                   this.container.views.createElementForEmbeddedExperience = this.handleNavigateEE();
 60                   this.container.views.destroyElement = this.handleDestroyElement();
 61                 },
 62                 
 63                 startup : function() {
 64                   this.gadgetToolbar = new GadgetToolbar({"gadgetArea" : this});
 65                   domConstruct.place(this.gadgetToolbar.domNode, this.domNode);
 66                   this.gadgetToolbar.startup();
 67                   var self = this;
 68                   this.gadgetToolbar.getPrefDialog().addPrefsChangedListener(function(prefs) {
 69                     var params = {};
 70                     params[osapi.container.RenderParam.USER_PREFS] = prefs;
 71                     self.reRenderGadget(params);
 72                   });
 73                   this.loadingWidget = new Loading();
 74                   domConstruct.place(this.loadingWidget.domNode, this.domNode);
 75                   this.loadingWidget.startup();
 76                   
 77                 },
 78                 
 79                 getContainer : function() {
 80                   return this.container;
 81                 },
 82                 
 83                 loadGadget : function(url) {
 84                   this.closeOpenSite();
 85                   this.loadingWidget.show();                  
 86                   var self = this;
 87                   this.container.preloadGadget(url, function(metadata) {
 88                     if(metadata[url] && !metadata[url].error) {
 89                       self.gadgetToolbar.setGadgetMetadata(metadata[url]);
 90                       self.renderGadget(url);
 91                     } else { 
 92                       console.error('There was an error fetching the metadata');
 93                     }
 94                   });
 95                 },
 96                 
 97                 renderGadget : function(url, opt_renderParams) {
 98                   var renderParams = opt_renderParams || {},
 99                       viewParams = {"gadgetUrl" : url};
100                   this.loadingWidget.show();                  
101                   this.site = this.createSite();
102                   renderParams[osapi.container.RenderParam.HEIGHT] = '100%';
103                   renderParams[osapi.container.RenderParam.WIDTH] = '100%';
104                   this.container.navigateGadget(this.site, url, viewParams, 
105                           renderParams);
106                 },
107                 
108                 setPrefs : function() {
109                   var self = this;
110                   return function(site, url, prefs) {
111                     self.gadgetToolbar.getPrefDialog().setPrefs(prefs);
112                   };
113                 },
114                 
115                 renderEmbeddedExperience : function(url, dataModel) {
116                   this.closeOpenSite();
117                   this.loadingWidget.show();  
118                   var self = this;
119                   this.container.preloadGadget(url, function(metadata) {
120                     if(metadata && metadata[url]) {
121                       self.gadgetToolbar.setGadgetMetadata(metadata[url]);
122                       var siteNode = self.createNodeForSite.call(self),
123                       oDataModel = JSON.parse(dataModel),
124                       renderParams = {};
125                       oDataModel.gadget = url;
126                       renderParams[osapi.container.ee.RenderParam.GADGET_RENDER_PARAMS] = {};
127                       renderParams[osapi.container.ee.RenderParam.GADGET_RENDER_PARAMS][osapi.container.RenderParam.HEIGHT] = '100%';
128                       renderParams[osapi.container.ee.RenderParam.GADGET_RENDER_PARAMS][osapi.container.RenderParam.WIDTH] = '100%';
129                       renderParams[osapi.container.ee.RenderParam.GADGET_VIEW_PARAMS] = {"gadgetUrl" : url}; 
130 
131                       self.container.ee.navigate(siteNode, oDataModel, renderParams, function(site){
132                         // Set the site so we can clean it up when we switch gadgets
133                         self.site = site;
134                       });
135                     } else {
136                       console.error('There was an error preloading the embedded experience.');
137                     }
138                   });
139                 },
140                 
141                 createSite : function() {
142                   var siteNode = this.createNodeForSite();
143                   return this.container.newGadgetSite(siteNode);
144                 },
145                 
146                 createNodeForSite: function() {
147                   // Let's be nice and reuse the same div for the site.
148                   var siteNode = dom.byId("gadgetSite" + this.siteCounter.toString());
149                   if (!siteNode) {
150                     this.siteCounter += 1;
151                     siteNode = domConstruct.create("div", {"id" : "gadgetSite" + this.siteCounter.toString()});
152                     domConstruct.place(siteNode, this.domNode);
153                   }
154                   return siteNode;
155                 },
156                 
157                 closeOpenSite: function() {
158                   if(this.site) {
159                     // IMPORTANT: The gadget must be unloaded before it is closed.
160                     // Otherwise, getUrl() is undefined and no lifecycle events are fired
161                     // for unload!!!
162                     this.container.unloadGadget(this.site.getActiveSiteHolder().getUrl());
163                     this.container.closeGadget(this.site);
164                     domConstruct.destroy("gadgetSite" + this.siteCounter.toString());
165                   }
166                 },
167                 
168                 reRenderGadget : function(opt_renderParams) {
169                   this.renderGadget(this.site.getActiveSiteHolder().getUrl(), opt_renderParams);
170                 },
171                 
172                 showActions : function(actionObjArray, gadgetToolbar, container, runAction) {
173                   for (var i = 0; i < actionObjArray.length; i++) {
174                     var action = actionObjArray[i];
175                     
176                     // Decorate the action with a function to be called when the action is executed
177                     if (action.path && action.path.length > 0) {
178                       action.runAction = function() {
179                         runAction(action.id);
180                       };
181                     } else if (action.dataType && action.dataType.length > 0) {
182                       action.runAction = function() {
183                         var selection = osData.get(action.dataType);
184                         container.selection.setSelection(selection);
185                         runAction(action.id);
186                       };
187                     } else {
188                       gadgets.error("Invalid action contribution: " + gadgets.json.stringify(action));
189                       break;
190                     }
191                     gadgetToolbar.addAction(action);
192                   }
193                 },
194                 
195                 hideActions : function(actionObjArray, gadgetToolbar) {
196                   for (var i = 0; i < actionObjArray.length; i++) {
197                     var action = actionObjArray[i];
198                     gadgetToolbar.removeAction(action);
199                   }
200                 },
201                 
202                 navigateForActions : function(gadgetArea, gadgetUrl, opt_params) {
203                   // We can effectively ignore gadgetUrl, because we'll get it from the site's holder in reRenderGadget
204                   gadgetArea.reRenderGadget(opt_params);
205                 },
206                 
207                 handleNavigateUrl : function() {
208                   var self = this;
209                   return function(rel, opt_viewTarget, opt_coordinates, parentSite, opt_callback) {
210                     return self.createDialog('URL', opt_viewTarget);
211                   };
212                 },
213                 
214                 handleNavigateGadget : function() {
215                   var self = this;
216                   return function (metadata, rel, opt_view, opt_viewTarget, opt_coordinates, parentSite, opt_callback) {
217                     var title = 'Gadget';
218                     if(metadata.modulePrefs && metadata.modulePrefs.title) {
219                       title = metadata.modulePrefs.title;
220                     }
221                     return self.createDialog(title, opt_viewTarget);
222                   };
223                 },
224                 
225                 handleNavigateEE : function() {
226                   var self = this;
227                   return function(rel, opt_gadgetInfo, opt_viewTarget, opt_coordinates, parentSite, opt_callback) {
228                     var title = 'Embedded Experiences';
229                     if(opt_gadgetInfo && opt_gadgetInfo.modulePrefs && opt_gadgetInfo.modulePrefs.title) {
230                       title = opt_gadgetInfo.modulePrefs.title;
231                     }
232                     return self.createDialog(title, opt_viewTarget);
233                   };
234                 },
235                 
236                 handleDestroyElement : function() {
237                   var self = this;
238                   return function(site) {
239                     self.gadgetDialog.hide(site);
240                   };
241                 },
242                 
243                 createDialog : function(title, viewTarget) {
244                   this.gadgetDialog = new GadgetModalDialog({"title" : title, "viewTarget" : viewTarget, 
245                     "container" : this.container});
246                   domConstruct.place(this.gadgetDialog.domNode, win.body());
247                   this.gadgetDialog.startup();
248                   this.gadgetDialog.show();
249                   return this.gadgetDialog.getGadgetNode();
250                 },
251                 
252                 destroy : function() {
253                   this.inherited(arguments);
254                   instance = undefined;
255                 }
256             });
257             
258             var instance;
259             
260             return {
261               getInstance : function() {
262                 if(!instance) {
263                   instance = new GadgetArea();
264                 }
265                 return instance;
266               }
267             };
268         });
