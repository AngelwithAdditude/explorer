 19 define(['dojo/_base/declare', 'dijit/_WidgetBase', 'dijit/_TemplatedMixin',
 20         'dojo/text!./../templates/MenuItem.html', 'dojo/dom-construct',
 21         'dojo/dom-class', 'modules/widgets/DropDownMenu', 'dojo/on'],
 22         function(declare, WidgetBase, TemplatedMixin,
 23                 template, domConstruct, domClass, DropDownMenu, on) {
 24             return declare('MenuItemWidget', [ WidgetBase, TemplatedMixin ], {
 25                 templateString : template,
 26                 
 27                 postCreate : function() {
 28                   
 29                 },
 30                 
 31                 startup : function() {
 32                   if(this.onclick) {
 33                     on(this.domNode,'click', this.onclick);
 34                   }
 35                   
 36                   if(this.menuItems) {
 37                     this.setSubMenuContent(this.menuItems.items, this.menuItems.direction);
 38                   }
 39                 },
 40                 
 41                 setSubMenuContent : function(menuItems, opt_direction) {
 42                   this.createSubMenu(opt_direction);
 43                   this.subMenu.clearMenuItems();
 44                   for(var i = 0; i < menuItems.length; i++) {
 45                     this.subMenu.addMenuItem(menuItems[i]);
 46                   }
 47                 },
 48                 
 49                 createSubMenu : function(opt_direction) {
 50                   if(!this.subMenu) {
 51                     this.subMenu = new DropDownMenu();
 52                     domConstruct.place(this.subMenu.domNode, this.domNode);
 53                     this.subMenu.startup();
 54                     domClass.add(this.domNode, 'dropdown-submenu');
 55                     if(opt_direction) {
 56                       domClass.add(this.domNode, opt_direction);
 57                     }
 58                   }
 59                 },
 60                 
 61                 getValue : function() {},
 62                 
 63                 addSubMenuItem : function(menuItem, opt_direction) {
 64                   this.createSubMenu(opt_direction);
 65                   this.subMenu.addMenuItem(menuItem);
 66                 },
 67                 
 68                 removeSubMenuItem : function(menuName) {
 69                   this.subMenu.removeMenuItem(menuName);
 70                 },
 71                 
 72                 getSubMenuItem : function(menuName) {
 73                   return this.subMenu.getMenuItem(menuName);
 74                 }
 75             });
 76         });
