 19 define(['dojo/_base/declare', 'modules/widgets/controlgroups/ControlGroup',
 20         'dojo/_base/array', 'dojo/query', 'dojo/dom-construct',
 21         'dojo/dom-attr', 'dojo/on', 'dojo/NodeList-manipulate', 'dojo/NodeList-dom'],
 22         function(declare, ControlGroup, arrayUtil, query, domConstruct, domAttr, on) {
 23             return declare('EnumControlGroupWidget', [ ControlGroup ], {
 24                 
 25                 postCreate : function() {
 26                   var select = domConstruct.create('select'),
 27                       self = this;
 28                   on(select, 'change', function(e) {
 29                     var options = e.currentTarget.options;
 30                     self.toggleSelection(options, e.currentTarget.value);
 31                   });
 32                   for(var i = 0; i < this.orderedEnumValues.length; i++) {
 33                     var option = domConstruct.create('option', 
 34                             {"value" : this.orderedEnumValues[i].value, "innerHTML" : this.orderedEnumValues[i].displayValue}, select);
 35                     if(this.orderedEnumValues[i].displayValue === this.defaultValue) {
 36                       domAttr.set(option, 'selected', 'selected');
 37                     }
 38                   }
 39                   domConstruct.place(select, query('.controls', this.domNode)[0]);
 40                   
 41                 },
 42                 
 43                 getValue : function() {
 44                   return this.currentValue;
 45                 },
 46                 
 47                 toggleSelection : function(options, selected) {
 48                   this.currentValue = selected;
 49                   arrayUtil.forEach(options, function(option) {
 50                     if(domAttr.get(option, 'selected')) {
 51                       domAttr.set(option, 'selected', '');
 52                     }
 53                     if(selected === domAttr.get(option, 'value')) {
 54                       domAttr.set(option, 'selected', 'selected');
 55                     }
 56                   });
 57                 },
 58                 
 59                 setValue : function(selected) {
 60                   var options = query('.controls select')[0].options;
 61                   this.toggleSelection(options, selected);
 62                 }
 63             });
 64         });
