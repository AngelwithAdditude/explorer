 19 define(['dojo/request/xhr', 'dojo/json'], function(xhr, json) {
 20   function getGadgetSpecBase(id, success, error) {
 21     xhr('gadgetspec/' + id, {
 22       handleAs: "json"
 23     }).then(function(data) {
 24       if(success) {
 25         success(data);
 26       }
 27     }, function(err) {
 28       if(error) {
 29         error(err);
 30       }
 31     });
 32   }
 33   return {
 34     getDefaultGadgetSpec : function(callbacks) {
 35       getGadgetSpecBase('default', callbacks.success, callbacks.error);
 36     },
 37     
 38     getGadgetSpec : function(id, callbacks) {
 39       getGadgetSpecBase(id, callbacks.success, callbacks.error);
 40     },
 41     
 42     createNewGadgetSpec : function(specData, callbacks) {
 43       xhr('gadgetspec', {
 44         handleAs: "json",
 45         method: "POST",
 46         data: json.stringify(specData),
 47         headers : {
 48           "Content-Type": "application/json"
 49         }
 50       }).then(callbacks.success, callbacks.error);
 51     },
 52     
 53     getSpecTree : function(callbacks) {
 54       xhr('gadgetspec/specTree', {
 55         handleAs: "json"
 56       }).then(callbacks.success, callbacks.error);
 57     }
 58   };
 59 });
