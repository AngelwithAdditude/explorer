 19 define(['dojo/_base/declare', 'dijit/_WidgetBase', 'dijit/_TemplatedMixin',
 20         'dojo/_base/array', 'dojo/text!./../../templates/EditorTab.html', 'dojo/dom-class',
 21         'dojo/on'],
 22         function(declare, WidgetBase, TemplatedMixin,
 23                 arrayUtil, template, domClass, on) {
 24             return declare('EditorTabWidget', [ WidgetBase, TemplatedMixin ], {
 25                 templateString : template,
 26                 
 27                 constructor : function() {
 28                   this.preActivationListeners = [];
 29                 },
 30                 
 31                 postCreate : function() {
 32                   if(this.isActive) {
 33                     this.activate();
 34                   }
 35                   var self = this;
 36                   on(this.domNode, 'click', function(e) {
 37                     for(var i = 0; i < self.preActivationListeners.length; i++) {
 38                       self.preActivationListeners[i]();
 39                     }
 40                     self.activate.call(self);
 41                   });
 42                 },
 43                 
 44                 activate : function() {
 45                   domClass.add(this.domNode, 'active');
 46                   this.editor.show();
 47                 },
 48                 
 49                 deactivate : function() {
 50                   domClass.remove(this.domNode, 'active');
 51                   this.editor.hide();
 52                 },
 53                 
 54                 addPreActivationListener : function(listener) {
 55                   this.preActivationListeners.push(listener);
 56                 },
 57                 
 58                 destroy : function() {
 59                   this.editor.destroy();
 60                   this.inherited(arguments);
 61                 },
 62                 
 63                 getResource : function() {
 64                   return {
 65                     "content" : this.editor.getContent(),
 66                     "name" : this.resource.name
 67                   };
 68                 },
 69                 
 70                 getEditor : function() {
 71                   return this.editor;
 72                 }
 73             });
 74         });
