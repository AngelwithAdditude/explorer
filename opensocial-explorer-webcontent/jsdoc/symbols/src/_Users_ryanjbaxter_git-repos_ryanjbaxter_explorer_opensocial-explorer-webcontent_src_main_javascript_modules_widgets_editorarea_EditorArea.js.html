 19 define([ 'dojo/_base/declare', 'dijit/_WidgetBase', 'dijit/_TemplatedMixin',
 20         'dojo/query', 'dojo/text!./../../templates/EditorArea.html', 'modules/widgets/editorarea/EditorToolbar',
 21         'modules/widgets/editorarea/EditorTabs', 'modules/widgets/editorarea/GadgetEditor', 'modules/widgets/editorarea/HtmlEditor', 
 22         'modules/widgets/editorarea/CssEditor', 'modules/widgets/editorarea/JSEditor', 'modules/widgets/editorarea/JSONEditor', 'modules/widgets/editorarea/EditorTab',
 23         'dojo/dom-construct', 'dojo/dom-class', 'modules/widgets/gadgetarea/GadgetArea',
 24         'modules/gadget-spec-service', 'dojo/NodeList-manipulate', 'dojo/NodeList-dom' ], 
 25         function(declare, WidgetBase, TemplatedMixin, query, template, EditorToolbar, EditorTabs, GadgetEditor, 
 26                 HtmlEditor, CssEditor, JSEditor, JSONEditor, EditorTab, domConstruct, domClass, GadgetArea,
 27                 gadgetSpecService) {
 28   var EditorArea =  declare('EditorAreaWidget', [ WidgetBase, TemplatedMixin ], {
 29           templateString : template,
 30           
 31           startup : function() {
 32             this.editorToolbar = new EditorToolbar({"editorArea" : this});
 33             domConstruct.place(this.editorToolbar.domNode, this.domNode);
 34             this.editorToolbar.startup();
 35             var self = this;
 36             this.getGadgetSpecService().getDefaultGadgetSpec({
 37               success : function(data) {
 38             	  self.addToUi.call(self, data);
 39             	},
 40             	error : function(data) {
 41             	  console.error("There was an error");
 42             	}
 43             });
 44           },
 45           
 46           displaySpec : function(id) {
 47             var self = this;
 48             this.getGadgetSpecService().getGadgetSpec(id, {
 49               success : function(data) {
 50                 self.editorTabs.removeAllTabs();
 51                 self.addToUi(data);
 52               },
 53               error : function(data) {
 54                 console.error("There was an error");
 55               }
 56             });
 57           },
 58           
 59           addToUi : function(data) {
 60             if(!this.editorTabs) {
 61               this.editorTabs = new EditorTabs(data);
 62             }
 63             domConstruct.place(this.editorTabs.domNode, this.domNode);
 64             this.editorTabs.startup();
 65             this.createTabsAndEditors(data, this.editorTabs);
 66             if (data.eeResource) {
 67               this.editorToolbar.showEEButton();
 68               this.renderEmbeddedExperience(data.id);
 69             } else {
 70               this.editorToolbar.hideEEButton();
 71               this.renderGadget(data.id);
 72             }
 73           },
 74           
 75           createTabsAndEditors : function(data, editorTabs) {
 76             var editor = new GadgetEditor({"resource" : data.gadgetResource});
 77             this.createTab(data.gadgetResource, editorTabs, true, editor);
 78             if(data.htmlResources) {
 79               for(var i = 0; i < data.htmlResources.length; i++) {
 80                 editor = new HtmlEditor({"resource" : data.htmlResources[i]});
 81                 this.createTab(data.htmlResources[i], editorTabs, false, editor);
 82               }
 83             }
 84             
 85             if(data.cssResources) {
 86               for(var i = 0; i < data.cssResources.length; i++) {
 87                 editor = new CssEditor({"resource" : data.cssResources[i]});
 88                 this.createTab(data.cssResources[i], editorTabs, false, editor);
 89               }
 90             }
 91             
 92             if(data.jsResources) {
 93               for(var i = 0; i < data.jsResources.length; i++) {
 94                 editor = new JSEditor({"resource" : data.jsResources[i]});
 95                 this.createTab(data.jsResources[i], editorTabs, false, editor);
 96               }
 97             }
 98             
 99             if(data.eeResource) {
100             	editor = new JSONEditor({"resource" : data.eeResource});
101             	this.createTab(data.eeResource, editorTabs, false, editor);
102             }
103             if(domClass.contains(this.domNode, 'topBottom')) {
104               query('.CodeMirror-scroll').addClass('topBottom');
105               editorTabs.refreshEditors();
106             }
107           },
108           
109           createTab : function(resource, editorTabs, isActive, editor) {
110             domConstruct.place(editor.domNode, this.domNode);
111             editor.startup();
112             var editorTab = new EditorTab({"resource" : resource, "isActive" : isActive, "editor" : editor});
113             editorTabs.addTab(editorTab);
114           },
115           
116           setTitle : function(title) {
117             if(this.editorToolbar) {
118               this.editorToolbar.setTitle(title);
119             }
120           },
121           
122           getGadgetSpec : function() {
123             return this.editorTabs.getGadgetSpec();
124           },
125           
126           renderGadget : function(id) {
127             GadgetArea.getInstance().loadGadget(document.location.protocol + '//' + document.location.host + this.getContextRoot() + '/gadgetspec/' + id + '/' + this.getGadgetSpec().gadgetResource.name);
128           },
129           
130           renderEmbeddedExperience : function(id) {
131             GadgetArea.getInstance().renderEmbeddedExperience(document.location.protocol + '//' + document.location.host + this.getContextRoot() + '/gadgetspec/' + id + '/' + this.getGadgetSpec().gadgetResource.name, 
132                     this.getGadgetSpec().eeResource.content);
133           },
134           
135           getContextRoot : function() {
136             var pathArray = document.location.pathname.split("/");
137             pathArray.pop();
138             return pathArray.join("/");
139           },
140           
141           getEditorTabs : function() {
142             return this.editorTabs;
143           },
144           
145           getGadgetSpecService : function() {
146             return gadgetSpecService;
147           },
148           
149           destroy : function() {
150             this.inherited(arguments);
151             instance = undefined;
152           }
153           
154   });
155   
156   var instance;
157   
158   return {
159     getInstance : function() {
160       if(!instance) {
161         instance = new EditorArea();
162       }
163       return instance;
164     }
165   };
166 });
