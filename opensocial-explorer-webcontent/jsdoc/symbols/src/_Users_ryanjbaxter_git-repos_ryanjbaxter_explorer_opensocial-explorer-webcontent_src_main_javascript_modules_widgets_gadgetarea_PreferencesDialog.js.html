 19 define(['dojo/_base/declare',  'modules/widgets/ModalDialog', 'dojo/query', 'dojo/dom-construct',
 20         'modules/widgets/controlgroups/StringControlGroup', 'modules/widgets/controlgroups/BooleanControlGroup', 'modules/widgets/controlgroups/EnumControlGroup', 
 21         'modules/widgets/controlgroups/ListControlGroup', 'dojo/on', 'dojo/NodeList-manipulate', 'dojo/NodeList-dom'],
 22         function(declare, ModalDialog, query, domConstruct, StringControlGroup, BooleanControlGroup, EnumControlGroup, ListControlGroup, on) {
 23             return declare('ModalDialogWidget', [ ModalDialog ], {
 24                 
 25                 constructor : function() {
 26                   this.controlGroups = {};
 27                   this.prefsChangedListeners = [];
 28                 },
 29                 
 30                 postCreate : function() {
 31                   var form = domConstruct.create('div', {"class" : "form-horizontal"}),
 32                       self = this,
 33                       closeBtn = domConstruct.create('button', {"class" : "btn", "innerHTML" : "Close"}),
 34                       saveBtn = domConstruct.create('button', {"class" : "btn btn-primary", "innerHTML" : "Save"}),
 35                       footer = query('.modal-footer', this.domNode);
 36                   query('.modal-body', this.domNode).append(form);
 37                   on(closeBtn, 'click', function(e) {
 38                     self.hide();
 39                   });
 40                   on(saveBtn, 'click', function(e) {
 41                     self.hide();
 42                     self.notifyPrefsChangedListeners.call(self);
 43                   });
 44                   footer.append(saveBtn);
 45                   footer.append(closeBtn);
 46                 },
 47                 
 48                 startup : function() {
 49                   this.setHeaderTitle('Preferences');
 50                   this.inherited(arguments);
 51                 },
 52                 
 53                 addPrefsToUI : function(prefs) {
 54                   query('.form-horizontal > *', this.domNode).remove();
 55                   //TODO we probably need to destroy the widgets created as well
 56                   for(var key in prefs) {
 57                     var pref = prefs[key];
 58                     this.addPrefToUI(pref);
 59                   }
 60                 },
 61                 
 62                 addPrefToUI : function(pref) {
 63                   var controlGroup;
 64                   if(pref.dataType === 'BOOL') {
 65                     controlGroup = new BooleanControlGroup(pref);
 66                   } else if(pref.dataType === 'ENUM') {
 67                     controlGroup = new EnumControlGroup(pref);
 68                   } else if(pref.dataType === 'STRING') {
 69                     controlGroup = new StringControlGroup(pref);
 70                   } else if(pref.dataType === 'LIST') {
 71                     controlGroup = new ListControlGroup(pref);
 72                   }
 73                   if(controlGroup) {
 74                     var form = query('.modal-body .form-horizontal', this.domNode);
 75                     form.append(controlGroup.domNode);
 76                     controlGroup.startup();
 77                     this.controlGroups[pref.name] = controlGroup;
 78                   }
 79                 },
 80                 
 81                 notifyPrefsChangedListeners : function() {
 82                   if(!this.isValid()) {
 83                     return;
 84                   }
 85                   var prefs = {};
 86                   for(var key in this.controlGroups) {
 87                     prefs[this.controlGroups[key].name] = this.controlGroups[key].getValue();
 88                   }
 89                   for(var j = 0; j < this.prefsChangedListeners.length; j++) {
 90                     this.prefsChangedListeners[j](prefs);
 91                   }
 92                 },
 93                 
 94                 isValid : function() {
 95                   //TODO go through and validate each field to make sure that the required prefs have values
 96                   return true;
 97                 },
 98                 
 99                 addPrefsChangedListener : function(listener) {
100                   this.prefsChangedListeners.push(listener);
101                 },
102                 
103                 setPrefs : function(prefs) {
104                   for(var key in prefs) {
105                     var controlGroup = this.controlGroups[key];
106                     if(controlGroup) {
107                       controlGroup.setValue(prefs[key]);
108                     }
109                   }
110                 }
111             });
112         });
