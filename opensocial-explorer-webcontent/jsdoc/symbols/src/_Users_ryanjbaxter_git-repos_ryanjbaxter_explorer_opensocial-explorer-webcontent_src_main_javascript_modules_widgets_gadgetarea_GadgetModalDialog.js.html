 19 define(['dojo/_base/declare',  'modules/widgets/ModalDialog', 'dojo/query', 
 20         'dojo/dom-class', 'modules/widgets/controlgroups/StringControlGroup', 'modules/widgets/controlgroups/BooleanControlGroup', 'modules/widgets/controlgroups/EnumControlGroup', 
 21         'modules/widgets/controlgroups/ListControlGroup', 'dojo/NodeList-manipulate', 'dojo/NodeList-dom'],
 22         function(declare, ModalDialog, query, domClass, StringControlGroup, BooleanControlGroup, EnumControlGroup, ListControlGroup) {
 23             return declare('GadgetModalDialogWidget', [ ModalDialog ], {
 24                 
 25                 postCreate : function() {
 26                   
 27                   if(this.viewTarget === 'TAB') {
 28                     domClass.add(this.domNode, 'tab');  
 29                     domClass.add(this.domNode, 'gadgetModal');
 30                   } else if( this.viewTarget === 'SIDEBAR') {
 31                     domClass.add(this.domNode, 'sidebar');
 32                     domClass.add(this.domNode, 'gadgetModal');
 33                   }
 34                 },
 35                 
 36                 startup : function() {
 37                   this.setHeaderTitle(this.title);
 38                   this.inherited(arguments);
 39                 },
 40                 
 41                 getGadgetNode : function() {
 42                   return query('.modal-body', this.domNode)[0];
 43                 },
 44                 
 45                 hide : function(opt_site) {
 46                   var site = opt_site;
 47                   if(!site) {
 48                     site = this.container.getGadgetSiteByIframeId_(query('.modal-body > iframe')[0].getAttribute('id'));
 49                   }
 50                   this.container.closeGadget(site);
 51                   this.inherited(arguments);
 52                   this.destroy();
 53                 }
 54             });
 55         });
