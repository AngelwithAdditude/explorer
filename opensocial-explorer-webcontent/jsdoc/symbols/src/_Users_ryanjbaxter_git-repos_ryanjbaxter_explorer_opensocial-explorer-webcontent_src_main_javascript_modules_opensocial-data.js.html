 19 define(['dojo/_base/declare'], function(declare) {
 20   var person = {
 21     "type" : "opensocial.Person",
 22     "dataObject" : {"id":"jane.doe","name":{"formatted":"Jane Doe","familyName":"Doe","givenName":"Jane"}}
 23   };
 24 
 25   var message = {
 26     "type" : "opensocial.Message",
 27     "dataObject" : {"id": "1", "senderId": "john.doe", "title": "Hairdo", "type": "publicMessage", "body": "nice "haircut!"", "replies": ["1a","1b"]}
 28   };
 29 
 30   file = {
 31     "type" : "opensocial.File",
 32     "dataObject" : {
 33       "author" : person,
 34       "displayName" : "My Presentation",
 35       "id" : "123",
 36       "mimeType" : "application/vnd.ms-powerpoint"
 37     }
 38   };
 39   return {
 40     get : function(type) {
 41       var data;
 42       if(type === 'opensocial.Person') {
 43         data = person;
 44       } else if(type === 'opensocial.Message') {
 45         data = message;
 46       } else if(type === 'opensocial.File') {
 47         data = file;
 48       }
 49       return data;
 50     }
 51   };
 52 });
